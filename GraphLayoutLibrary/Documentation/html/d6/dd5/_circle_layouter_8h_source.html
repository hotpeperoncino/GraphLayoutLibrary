99  /*

104  LAYOUT_ASSERT(&gGraph != NULL, LayoutMemoryException(__FUNCTION__, LayoutExceptionEnum::NULL_POINTER_EXCEPTION, NULL_GRAPH_FOUND));
105  LAYOUT_ASSERT(radius > 0, LayoutException(__FUNCTION__, LayoutExceptionEnum::INVALID_PARAMETER, INVALID_RADIUS_VALUE, ""));

107 
108  // For every vertex, Calculate x and y coordinates
109  size_t iOrderValue = 1;
110 
111  IteratorMapOrderVertex mapIterOrderVertex(mapOrderVertex);
112 
113  double dAngleSpace = 0;
114  SizeManager sizeManager;
115 
116  double dPrevNodeSpace = 0;
117 
118  // Iterate vertices in ordered mannar
119  while(mapIterOrderVertex.hasNext())
120  {
121  mapIterOrderVertex.next();
122  VertexDescriptor vVertex = mapIterOrderVertex.value();
123 
124  // Calculate angle space for this node
125  double dAngleSpacePresentNode;
126  try
127  {
128  dAngleSpacePresentNode = calculateAngleSpace(vVertex, gGraph);
129  }
130  catch(LayoutMemoryException& eException)
131  {
132  throw LayoutMemoryException(__FUNCTION__, LayoutExceptionEnum::NULL_POINTER_EXCEPTION, eException.getObjectName());
133  }

137  qDebug()<<"\nDefault Angle space caluclated for vertex " << sVertexId;
138  int iTotalVertices = num_vertices(gGraph);
139  dAngleSpacePresentNode = ((2 * PI) / iTotalVertices);
140  }

213  {
214  // For every vertex, Calculate x and y coordinates
215  // VertexSizeType vertexCount = num_vertices(gGraph);

217 
218  VertexSizeType indexCount = 0;
219  double dDegreeToRadian = 0.0174532925;
220  double dRadianToDegree = 57.32484076433121;
221  double dAngleSpace;
222 
223  if(dStartAngle > dEndAngle)
224  {
225  dAngleSpace = ((((360 - dStartAngle) + dEndAngle) * dDegreeToRadian) / vertexCount);
226  }
227  else

229  dAngleSpace = ((abs(dEndAngle - dStartAngle) * dDegreeToRadian) / vertexCount);
230  }
231 
232  IteratorMapOrderVertex mapIter(mapOrderVertex);

239  mapIter.next();

242  double dCosValue = cos(dStartAngle + (indexCount * dAngleSpace));
243  double dSInValue = sin(dStartAngle + (indexCount * dAngleSpace));
